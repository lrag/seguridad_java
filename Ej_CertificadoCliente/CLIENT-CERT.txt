
-crear el certificado del cliente

jks

keytool -genkey -alias client-alias -keyalg RSA -keypass changeit -storepass changeit -keystore client_keystore.jks

pkcs12

keytool -genkey -alias client-alias -keyalg RSA -keypass changeit -storepass changeit -keystore client_keystore.pfx -storetype PKCS12  


-crear el certificado para importarlo en el so/navegador

keytool -importkeystore -srckeystore .\client_keystore.pfx -srcstorepass changeit -srcalias client-alias -destkeystore .\client.p12 -deststoretype PKCS12 -deststorepass changeit -destalias destalias -noprompt


-crear el truststore para el tomcat

keytool -import -alias alias -file client.cer -storetype PKCS12 -keystore server.truststore
